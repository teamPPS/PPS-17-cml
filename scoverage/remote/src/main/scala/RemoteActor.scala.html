18 class cml.RemoteActor extends Actor with ActorLogging {
19   val DefaultMessage: String = "WARNING: cml.RemoteActor has not receive anything"
20   var actorInList = new ListBuffer[ActorRef]
21   var isFirst: Boolean = true
22 
23   override def receive: Receive = {
24     case RequireEnterInArena() =>
25       addIntoBattleUserList(sender)
26       sender ! RequireEnterInArenaSuccess()
27     case ExistChallenger() =>
28       val exist = existChallenger()
29       if (exist) actorInList.foreach(actor => actor ! ExistChallengerSuccess(actorInList))
30     case ExitRequest() =>
31       removeIntoBattleUserList(sender)
32     case RequireTurnRequest(attackPower, turn) =>
33       turnManagement(turn)
34       sender ! RequireTurnSuccess(attackPower, turn)
35     case _ => log.info(DefaultMessage)
36   }
37 
38   private def addIntoBattleUserList(actorIdentity: ActorRef){
39     actorInList += actorIdentity
40     log.info("LIST add --> " + userList_ )
41   }
42 
43   private def removeIntoBattleUserList(actorIdentity: ActorRef) {
44     actorInList -= actorIdentity
45     log.info("LIST remove --> " + userList_)
46   }
47 
48   //todo: technical debit
49   private def existChallenger(): Boolean = {
50     if (userList_().length.equals(2)) true
51     else false
52   }
53 
54   private def userList_(): ListBuffer[ActorRef] ={
55     actorInList
56   }
57 
58   private def turnManagement(turn: Int): Unit = {
59     if(isFirst && initializeTurn(turn)) isFirst = false
60     changeTurn(turn)
61   }
62 
63   val First: Int = 0
64   val Second: Int = 1
65 
66   private def initializeTurn(turn: Int): Boolean = {
67     turn.equals(First)
68   }
69 
70   private def changeTurn(turn: Int): Int = {
71     var turn_ : Int = First
72     if (turn equals First) turn_ = Second
73     else turn_ = First
74     turn_
75   }
76 }
