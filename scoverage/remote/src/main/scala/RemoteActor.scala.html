32   override def receive: Receive = {
33     case RequireEnterInArena() =>
34       addIntoBattleUserList(sender)
35       sender ! RequireEnterInArenaSuccess()
36     case ExistChallenger() =>
37       val exist = existChallenger()
38       if (exist) actorList.foreach{ actor => actor ! ExistChallengerSuccess(actorList) }
39     case ExitRequest() =>
40       removeIntoBattleUserList(sender)
41     case RequireTurnRequest(attackPower, isProtected, turn) =>
42       sender ! RequireTurnSuccess(attackPower, isProtected, turnManagement.changeTurn(turn))
43     case _ => stash()
44   }
45 
46   private def addIntoBattleUserList(actorIdentity: ActorRef){
47     actorList += actorIdentity
48     tmpUserList += actorIdentity
49     log.info("User in list (add option) -> " + userList_())
50   }
51 
52   private def removeIntoBattleUserList(actorIdentity: ActorRef) {
53     actorList -= actorIdentity
54     log.info("User in list (remove option) -> " + userList_)
55     unstashAll()
56   }
57 
58   private def existChallenger(): Boolean = {
59     if (userList_().size.equals(2)) true
60     else false
61   }
62 
63   private def userList_():  ListBuffer[ActorRef]  = {

65   }
