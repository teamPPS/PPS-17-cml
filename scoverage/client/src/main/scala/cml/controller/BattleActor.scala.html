42   override def receive: Receive = {
43     case SceneInfo(scene) => sceneContext = scene
44     case RequireEnterInArenaSuccess() =>
45       remoteActor ! ExistChallenger()
46     case ExistChallengerSuccess(user) =>
47       remoteActor ! ExitRequest()
48       log.info("User in list - " + user)
49       myChallenge(user)
50       self ! SwitchInArenaRequest()
51     case SwitchInArenaRequest() =>
52       arenaActor ! ActorRefRequest(self)
53       Platform.runLater(() => switchInArena())
54     case AttackRequest(attackPower) => remoteActor ! RequireTurnRequest(attackPower, turn)
55     case RequireTurnSuccess(attackPower, turnValue) =>
56       log.info("Turn" + turnValue)
57       if(turnValue equals turn) challenger ! AttackSuccess(attackPower)
58     case AttackSuccess(attackPower) =>
59       log.info("Attack " + attackPower)
60       arenaActor ! AttackSuccess(attackPower)
61     case StopRequest() => context.stop(self)
62   }
63 
64   private def myChallenge(user: ListBuffer[ActorRef]): Unit = {
65     user.foreach(actor => if(!actor.equals(self)) challenger = actor)
66     log.info("Im user: " + self + " and my challenger is - " + challenger)
67     _turn(user)
68   }
69 
70   private def _turn(user: ListBuffer[ActorRef]): Unit = {
71     if(user.head equals self) turn = 0
72     else turn = 1
73     log.info("My turn is: " + turn)
74   }
75 
76   private def switchInArena(): Unit ={
77     ViewSwitch.activate(ArenaWindow.path, sceneContext)
78   }
