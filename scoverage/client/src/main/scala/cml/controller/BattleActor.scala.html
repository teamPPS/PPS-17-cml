34   @throws[Exception](classOf[Exception])
35   override def preStart(): Unit = {
36     remoteActor = context.actorSelection(RemoteActorInfo.Path)
37     arenaActor = system.actorOf(Props(new ArenaActor()), "ArenaActor")

39   }

41   override def postStop(): Unit = {
42     log.info("Actor is stopped")
43   }

45   override def receive: Receive = {
46     case SceneInfo(scene) => sceneContext = scene
47     case RequireEnterInArenaSuccess() => remoteActor ! ExistChallenger()
48     case ExistChallengerSuccess(user) =>
49       remoteActor ! ExitRequest()
50       challenger_(user)
51     case CreatureRequire(creature) =>
52       challengerCreature = creature
53       self ! SwitchInArenaRequest()
54     case SwitchInArenaRequest() =>
55       arenaActor ! ActorRefRequest(self, challenger, challengerCreature, turn)
56       Platform.runLater(() => switchInArena())
57     case AttackRequest(attackPower, protection) => remoteActor ! RequireTurnRequest(attackPower, protection, turn)
58     case RequireTurnSuccess(attackPower, protection, turnValue) =>
59       log.info("Turn" + turnValue)
60       challenger ! AttackSuccess(attackPower, protection, turnValue)
61     case AttackSuccess(attackPower, isProtected, turnValue) =>
62       log.info("Attack " + attackPower + " is protected " + isProtected)
63       arenaActor ! AttackSuccess(attackPower, isProtected, turnValue)
64     case StopRequest(scene) =>
65       arenaActor ! NotifierExit(self)
66       Platform.runLater(() => ViewSwitch.activate(VillageWindow.path, scene))
67       context.stop(self)
68     case NotifierExitSuccess() => arenaActor ! NotifierExitSuccess()
69     case _ =>
70   }
71 
72   private def challenger_(user: ListBuffer[ActorRef]): Unit = {
73     user.foreach{ actor => if(!actor.equals(self)) challenger = actor}
74     challenger ! CreatureRequire(selectedCreature)
75     log.info("Im user: " + self + " and my challenger is - " + challenger)
76     _turn(user)
77   }
78 
79   private def _turn(user:  ListBuffer[ActorRef]): Unit = {
80     if(user.head equals self) turn = 0
81     else turn = 1
82     log.info("My turn is: " + turn)
83   }
84 
85   private def switchInArena(): Unit ={
86     ViewSwitch.activate(ArenaWindow.path, sceneContext)
87   }
88 }
