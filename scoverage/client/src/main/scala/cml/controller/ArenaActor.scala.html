27   override def receive: Receive = {
28     case ActorRefRequest(actor, challenger,creature, turnValue) =>
29       battleActor = actor
30       challengerCreature = creature
31       turn = turnValue
32       tmpUserList += actor
33       tmpUserList += challenger
34     case ControllerRefRequest(controllerValue) =>
35       controller = controllerValue
36       Platform.runLater(() => controller.turn_(turn))
37     case ChallengerCreatureRequire() => Platform.runLater(() => controller.challengeCreature(challengerCreature))
38     case AttackRequest(value, protection) => battleActor ! AttackRequest(value, protection)
39     case AttackSuccess(value, isProtected, turnValue) =>
40       Platform.runLater(() => controller.userLifeBar_(value, isProtected, turnValue))
41     case NotifierExit(actor) =>
42       tmpUserList -= actor
43       tmpUserList.foreach( actorInList => actorInList ! NotifierExitSuccess())
44       if(actor equals battleActor) context.stop(self)
45     case NotifierExitSuccess() =>
46       Platform.runLater(() => controller.exitOption())
47     case StopRequest(scene) =>
48       battleActor ! StopRequest(scene)
49   }
