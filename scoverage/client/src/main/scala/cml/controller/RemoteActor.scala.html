20 class cml.controller.cml.cml.RemoteActor extends Actor with ActorLogging {
21   val DefaultMessage: String = "WARNING: cml.controller.cml.cml.RemoteActor has not receive anything"
22   var actorInList = new ListBuffer[ActorRef]
23   var isFirst: Boolean = true
24 
25   override def receive: Receive = {
26     case RequireEnterInArena() =>
27       addIntoBattleUserList(sender)
28       sender ! RequireEnterInArenaSuccess()
29     case ExistChallenger() =>
30       val exist = existChallenger()
31       if (exist) actorInList.foreach(actor => actor ! ExistChallengerSuccess(actorInList))
32     case ExitRequest() =>
33       removeIntoBattleUserList(sender)
34     case RequireTurnRequest(attackPower, turn) =>
35       turnManagement(turn)
36       sender ! RequireTurnSuccess(attackPower, turn)
37     case _ => log.info(DefaultMessage)
38   }
39 
40   private def addIntoBattleUserList(actorIdentity: ActorRef){
41     actorInList += actorIdentity
42     log.info("LIST add --> " + userList_ )
43   }
44 
45   private def removeIntoBattleUserList(actorIdentity: ActorRef) {
46     actorInList -= actorIdentity
47     log.info("LIST remove --> " + userList_)
48   }
49 
50   //todo: technical debit
51   private def existChallenger(): Boolean = {
52     if (userList_().length.equals(2)) true
53     else false
54   }
55 
56   private def userList_(): ListBuffer[ActorRef] ={
57     actorInList
58   }
59 
60   private def turnManagement(turn: Int): Unit = {
61     if(isFirst && initializeTurn(turn)) isFirst = false
62     changeTurn(turn)
63   }
64 
65   val First: Int = 0
66   val Second: Int = 1
67 
68   private def initializeTurn(turn: Int): Boolean = {
69     turn.equals(First)
70   }
71 
72   private def changeTurn(turn: Int): Int = {
73     var turn_ : Int = First
74     if (turn equals First) turn_ = Second
75     else turn_ = First
76     turn_
77   }
78 }
