26 
27   val authenticationVertx = AuthenticationServiceVertxImpl()
28   val villageActor: ActorRef = context.system.actorOf(Props[VillageActor], "VillageActor")
29   var authController: AuthenticationViewController = controller
30 
31   override def receive: Receive = authenticationBehaviour orElse villageBehaviour
32 
33   private def authenticationBehaviour: Receive = {
34     case Register(username, password) => authenticationVertx.register(username, password)
35       .onComplete {
36         case Success(httpResponse) =>
37           if(checkHttpResponse(httpResponse, registerSuccess, registerFailure))
38             villageActor ! CreateVillage()
39         case Failure(exception) =>
40           RegisterFailure(exception.getMessage)
41           displayMsg(registerFailure)
42       }
43     case Login(username, password) => authenticationVertx.login(username, password)
44       .onComplete {
45         case Success(httpResponse) =>
46           if(checkHttpResponse(httpResponse, loginSuccess, loginFailure))
47             loginSucceedOnGui()
48         case Failure(exception) =>
49           LoginFailure(exception.getMessage)
50           displayMsg(loginFailure)
51       }
52     case Logout() => authenticationVertx.logout(TokenStorage.getUserJWTToken)
53     case SetController(controller: AuthenticationViewController) =>  authController = controller
54   }
55 
56   private def villageBehaviour: Receive = {
57     case CreateVillageSuccess() =>
58       log.info("Village created with success")
59       loginSucceedOnGui()
60     case VillageFailure(m) => displayMsg(m)
61   }
62 
63   def displayMsg(m: String): Unit = Platform.runLater(() => authController.formMsgLabel.setText(m))
64 
65   def loginSucceedOnGui(): Unit = Platform.runLater(() => authController.openVillageView())
66 
67   def successAuthenticationCase(tokenValue: String): Unit = {
68     log.info("TOKEN: " + tokenValue)
69     TokenStorage.setUserJWTToken(tokenValue)
70   }
71 
72   def checkHttpResponse(str: String, successMessage: String, failureMessage: String): Boolean = str match { //technical debt?
73     case "Not a valid request" =>
74       displayMsg(failureMessage)
75       false
76     case _ =>
77       successAuthenticationCase(str)
78       displayMsg(successMessage)
79       log.info("Success this is server response with the token: " + str)
80       true
81   }
82   
83   def disableButtons(b: Boolean): Unit = {
84     authController.registerButton.setDisable(b)
85     authController.loginButton.setDisable(b)
86   }
