26 
27   val authenticationVertx = AuthenticationServiceVertxImpl()
28   val villageActor: ActorRef = context.system.actorOf(Props(new VillageActor()), "VillageActor")
29 
30   override def receive: Receive = authenticationBehaviour orElse villageBehaviour
31 
32   /**
33     * @return the authentication behaviour
34     */
35   private def authenticationBehaviour: Receive = {
36     case Register(username, password) => authenticationVertx.register(username, password)
37       .onComplete {
38         case Success(httpResponse) =>
39           checkHttpResponse(httpResponse, loginSuccess, loginFailure)
40           villageActor ! CreateVillage()
41         case Failure(exception) =>
42           RegisterFailure(exception.getMessage)
43           displayMsg(registerFailure)
44       }
45     case Login(username, password) => authenticationVertx.login(username, password)
46       .onComplete {
47         case Success(httpResponse) =>
48           if(checkHttpResponse(httpResponse, loginSuccess, loginFailure))
49             loginSucceedOnGui()
50         case Failure(exception) =>
51           LoginFailure(exception.getMessage)
52           displayMsg(loginFailure)
53       }
54     case Logout() => authenticationVertx.logout(TokenStorage.getUserJWTToken)
55   }
56 
57   /**
58     * @return behaviour when creating and entering a village
59     */
60   private def villageBehaviour: Receive = {
61     case CreateVillageSuccess() =>
62       println("Village create success")
63       loginSucceedOnGui()
64     case VillageFailure(m) => displayMsg(m)
65   }
66 
67   /**
68     * Displays text on the GUI through a label
69     *
70     * @param m message to show
71     */
72   def displayMsg(m: String): Unit = Platform.runLater(() => controller.formMsgLabel.setText(m))
73 
74   /**
75     * Open Village View after login successful
76     */
77   def loginSucceedOnGui(): Unit = Platform.runLater(() => controller.openVillageView())
78 
79   def successAuthenticationCase(tokenValue: String): Unit = {
80     println("TOKEN: " + tokenValue)
81     TokenStorage.setUserJWTToken(tokenValue)
82   }
83 
84   def checkHttpResponse(str: String, successMessage: String, failureMessage: String): Boolean = str match { //technical debt?
85     case "Not a valid request" =>
86       displayMsg(failureMessage)
87       false
88     case _ =>
89       successAuthenticationCase(str)
90       displayMsg(successMessage)
91       println("Success this is server response with the token: " + str)
92       true
93   }
94 
95   /**

102     controller.loginButton.setDisable(b)
103   }
