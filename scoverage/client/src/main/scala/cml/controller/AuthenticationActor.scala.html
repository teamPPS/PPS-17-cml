25 class AuthenticationActor(controller: AuthenticationViewController) extends Actor {
26 
27   val authenticationVertx = AuthenticationServiceVertxImpl()
28   val villageActor: ActorRef = context.system.actorOf(Props(new VillageActor()), "VillageActor")
29   var authController = controller
30 
31   override def receive: Receive = authenticationBehaviour orElse villageBehaviour
32 
33   /**
34     * @return the authentication behaviour
35     */
36   private def authenticationBehaviour: Receive = {
37     case Register(username, password) => authenticationVertx.register(username, password)
38       .onComplete {
39         case Success(httpResponse) =>
40           if(checkHttpResponse(httpResponse, loginSuccess, loginFailure))
41             villageActor ! CreateVillage()
42         case Failure(exception) =>
43           RegisterFailure(exception.getMessage)
44           displayMsg(registerFailure)
45       }
46     case Login(username, password) => authenticationVertx.login(username, password)
47       .onComplete {
48         case Success(httpResponse) =>
49           if(checkHttpResponse(httpResponse, loginSuccess, loginFailure))
50             loginSucceedOnGui()
51         case Failure(exception) =>
52           LoginFailure(exception.getMessage)
53           displayMsg(loginFailure)
54       }
55     case Logout() => authenticationVertx.logout(TokenStorage.getUserJWTToken)
56     case SetController(controller: AuthenticationViewController) =>  authController = controller
57   }
58 
59   /**
60     * @return behaviour when creating and entering a village
61     */
62   private def villageBehaviour: Receive = {
63     case CreateVillageSuccess() =>
64       println("Village create success")
65       loginSucceedOnGui()
66     case VillageFailure(m) => displayMsg(m)
67   }
68 
69   /**
70     * Displays text on the GUI through a label
71     *
72     * @param m message to show
73     */
74   def displayMsg(m: String): Unit = Platform.runLater(() => authController.formMsgLabel.setText(m))
75 
76   /**
77     * Open Village View after login successful
78     */
79   def loginSucceedOnGui(): Unit = Platform.runLater(() => authController.openVillageView())
80 
81   def successAuthenticationCase(tokenValue: String): Unit = {
82     println("TOKEN: " + tokenValue)
83     TokenStorage.setUserJWTToken(tokenValue)
84   }
85 
86   def checkHttpResponse(str: String, successMessage: String, failureMessage: String): Boolean = str match { //technical debt?
87     case "Not a valid request" =>
88       displayMsg(failureMessage)
89       false
90     case _ =>
91       successAuthenticationCase(str)
92       displayMsg(successMessage)
93       println("Success this is server response with the token: " + str)
94       true
95   }
96 
97   /**
98     * Switches on and off GUI buttons
99     *
100     * @param b boolean
101     */
102   def disableButtons(b: Boolean): Unit = {
103     authController.registerButton.setDisable(b)
104     authController.loginButton.setDisable(b)
105   }
