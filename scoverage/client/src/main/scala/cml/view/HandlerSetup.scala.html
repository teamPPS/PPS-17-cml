104   private def addDragAndDropSourceHandler(t: Tile, c: VillageViewController): Unit = {
105     val canvas = t.imageSprite

108       val dragBoard: Dragboard = canvas startDragAndDrop TransferMode.COPY

121   private def addDragAndDropTargetHandler(n: Node, c: VillageViewController): Unit = {
122     n setOnDragOver ((event: DragEvent) => {

127     n setOnDragDropped ((event: DragEvent) => {
128       val gold = VillageMap.instance().get.gold
129       if(enoughResourceForBuild(gold)){
130         val dragBoard: Dragboard = event getDragboard()
131         val newTile = tileSet.filter(t => t.description.equals(dragBoard.getString)).head
132         n match {
133           case i: ImageView => i setImage newTile.imageSprite.snapshot(new SnapshotParameters, null)
134           case _ => throw new ClassCastException
135         }
136         val y = GridPane.getColumnIndex(n)
137         val x = GridPane.getRowIndex(n)
138         if(!VillageMap.instance().get.villageStructure.exists(s => s.position.equals(Position(x, y)))) {
139           updateVillage(newTile, x, y)
140           decrementMoney(gold, Price, c)
141           c.selectionInfo setText DroppedMessage(dragBoard.getString, x, y)
142         } else {
143           c.selectionInfo setText CantBuildInOldOne
144         }
145       } else c.selectionInfo setText NotHaveMoney
146       event consume()
147     })

158   private def upgradeStructure(s: Structure, c: VillageViewController): Unit = {
159     val gold =  VillageMap.instance().get.gold
160     val food =  VillageMap.instance().get.food
161     if(enoughResourceForLevelUp(gold, s.level)) {
162       s.creatures match {
163         case None => {
164           upgradeBuildings(s)
165           decrementMoney(gold, Price*s.level, c)
166           c.selectionInfo setText displayText(getClassName(s), s.level, s.resource.amount, s.creatures)
167         }
168         case _ =>
169           if(enoughResourceForLevelUp(food, s.level)) {
170             upgradeHabitat(s)
171             decrementFood(food, Price * s.level, c)
172             decrementMoney(gold, Price*s.level, c)
173             c.selectionInfo setText displayText(getClassName(s), s.level, s.resource.amount, s.creatures)
174           } else {
175             c.selectionInfo setText NotHaveFood
176           }
177       }
178     } else {
179       c.levelUpButton setDisable true
180       c.selectionInfo setText NotHaveMoney
181     }
