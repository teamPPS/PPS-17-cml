93   private def addDragAndDropSourceHandler(t: Tile, c: VillageViewController): Unit = {
94     val canvas = t.imageSprite
95     canvas setOnMouseClicked (_ => c.selectionInfo setText "Element selected: "+ t.description + "\nPrice: "+price)
96     canvas setOnDragDetected ((event: MouseEvent) => {
97       val dragBoard: Dragboard = canvas startDragAndDrop TransferMode.COPY

100       val content: ClipboardContent = new ClipboardContent

103       val gold = VillageMap.instance().get.gold
104       if(gold >= price) c.selectionInfo setText "Dragged element " + dragBoard.getString

111   private def addDragAndDropTargetHandler(n: Node, c: VillageViewController): Unit = {
112     n setOnDragOver ((event: DragEvent) => {
113       event acceptTransferModes TransferMode.COPY
114       event consume()
115     })
116 
117     n setOnDragDropped ((event: DragEvent) => {
118       val gold = VillageMap.instance().get.gold
119       if(gold >= price){
120         val dragBoard: Dragboard = event getDragboard()
121         val newTile = tileSet.filter(t => t.description.equals(dragBoard.getString)).head
122         n match {
123           case i: ImageView => i setImage newTile.imageSprite.snapshot(new SnapshotParameters, null)
124           case _ => throw new ClassCastException
125         }
126         val y = GridPane.getColumnIndex(n)
127         val x = GridPane.getRowIndex(n)
128 
129         val structure = StaticStructure(newTile, x, y)
130         val json = structure.json
131         VillageMap.instance().get.villageStructure += structure.getStructure
132         villageActor ! UpdateVillage(json)
133         c.selectionInfo setText "Dropped element " + dragBoard.getString + " in coordinates (" + x + " - " + y + ")"
134 
135         decrementMoney(gold, price, c)
136 
137       } else c.selectionInfo setText "You can't build a structure if you don't have money"
138       event consume()
139     })
140   }

150   private def upgradeStructure(s: Structure, c: VillageViewController): Unit = {
151     val gold =  VillageMap.instance().get.gold
152     val food =  VillageMap.instance().get.food
153     if(gold >= price){
154       val upgrade = StructureUpgrade(s)
155       upgrade creatureJson match {
156         case null => villageActor ! SetUpdateVillage(upgrade structureJson)
157         case _ => {
158           villageActor ! SetUpdateVillage(upgrade creatureJson)
159           decrementFood(food, price*s.level, c)
160         }
161       }
162       decrementMoney(gold, price*s.level, c)
163       c.selectionInfo setText displayText(getClassName(s), s.level, s.resource.amount, s.creatures)
164     }
165     else{
166       c.levelUpButton setDisable true
167       c.selectionInfo setText "You can't upgrade a structure if you don't have money"
168     }
