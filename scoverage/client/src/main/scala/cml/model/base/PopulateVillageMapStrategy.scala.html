45       creatureType <- recursiveReadJsonField(creature, Village.CREATURE_TYPE_FIELD);
46       specificCreature = creatureType.as[String] match {
47         case ModelConfig.Creature.DRAGON => creature.as[FireCreature]

49         case ModelConfig.Creature.GRIFFIN => creature.as[AirCreature]
50         case ModelConfig.Creature.WATERDEMON => creature.as[WaterCreature]
51       }
52     ) yield {

54       VillageMap.instance().get.villageStructure += specificHabitat
55     }
56 
57     for (
58       habitat <- habitats
59       if !jsonFieldExist(habitat, Village.SINGLE_CREATURE_FIELD);
60       specificHabitat = habitat.as[Habitat.Habitat]
61     ) yield {
62       VillageMap.instance().get.villageStructure += specificHabitat
63     }
64   }
