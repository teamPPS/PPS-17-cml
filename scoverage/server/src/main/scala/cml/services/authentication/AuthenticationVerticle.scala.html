47   private def login: Handler[RoutingContext] = implicit routingContext => {
48     println("Receive login request")
49     (for (
50       headerAuthorization <- getRequestAndHeader;
51       (username, password) <- TokenAuthentication.checkBase64Authentication(headerAuthorization)
52     ) yield {
53       authenticationService.login(username, password).onComplete {
54         case Success(x) =>
55           if(x) JWTAuthentication.encodeUsernameToken(username).foreach(sendResponse(OK,_)) else {
56             sendResponse(UNAUTHORIZED, UNAUTHORIZED.toString)

58         case Failure(_) => sendResponse(UNAUTHORIZED, UNAUTHORIZED.toString)
59       }
60     }).getOrElse(sendResponse(BAD_REQUEST, BAD_REQUEST.toString))
61   }
