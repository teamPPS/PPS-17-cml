36   private def register: Handler[RoutingContext] = implicit routingContext => {
37     log.info("Receive register request", None)
38     (for(
39       request <- getRequestAndHeader;

50   private def login: Handler[RoutingContext] = implicit routingContext => {
51     log.info("Receive login request", None)
52     (for (
53       headerAuthorization <- getRequestAndHeader;
54       (username, password) <- TokenAuthentication.checkBase64Authentication(headerAuthorization)
55     ) yield {
56       authenticationService.login(username, password).onComplete {
57         case Success(x) =>
58           if(x) JWTAuthentication.encodeUsernameToken(username).foreach(sendResponse(OK,_)) else {
59             sendResponse(UNAUTHORIZED, UNAUTHORIZED.toString)
60           }
61         case Failure(_) => sendResponse(UNAUTHORIZED, UNAUTHORIZED.toString)
62       }
63     }).getOrElse(sendResponse(BAD_REQUEST, BAD_REQUEST.toString))
64   }
