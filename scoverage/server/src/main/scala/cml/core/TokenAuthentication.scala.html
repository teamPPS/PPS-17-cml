46     for(

48       checkPassword <- Option(password) if checkPassword.nonEmpty

50   }

52   override def checkBase64Authentication(header: String): Option[(String, String)] = {
53     for (
54       checkHeader <- Option(header) if checkHeader.nonEmpty;
55       headerInfo = checkHeader.split(s"$BASE64")(1) if headerInfo.nonEmpty;
56       headerDecode = new String(Base64.getDecoder.decode(headerInfo)).split(":")
57     ) yield (headerDecode(0), headerDecode(1))
58   }
