51   override def base64Authentication(username: String, password: String): Option[String] = {
52     for(
53       checkUsername <- Option(username) if checkUsername.nonEmpty;
54       checkPassword <- Option(password) if checkPassword.nonEmpty
55     ) yield s"$BASE64" + Base64.getEncoder.encodeToString(s"$checkUsername:$checkPassword".getBytes())
56   }
57 
58   override def checkBase64Authentication(header: String): Option[(String, String)] = {
59     for (
60       checkHeader <- Option(header) if checkHeader.nonEmpty;
61       headerInfo = checkHeader.split(s"$BASE64")(1) if headerInfo.nonEmpty;
62       headerDecode = new String(Base64.getDecoder.decode(headerInfo)).split(":")
63     ) yield (headerDecode(0), headerDecode(1))
64   }
